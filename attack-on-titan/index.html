// 시간 기반 무작위 재난 스폰 매니저 — 라운드별 다른 패턴
enum DisasterType {
    NORMAL_TITAN     = 0,
    BEAST_TITAN_THROW = 1,   // v2 · 짐승 거인 투석
    FEMALE_TITAN_HIT  = 2,   // v2 · 여성형 거인 공격
    DEBRIS_FALL       = 3,
}

interface IDisaster {
    spawn(position: Vector3): void;
    tick(deltaTime: number): void;
    despawn(): void;
}

export class DisasterManager {
    private elapsed: number = 0;
    private active: IDisaster[] = [];
    private schedule: DisasterEvent[];   // 가중치 + 시간 슬롯 테이블

    // 매 프레임 호출은 비용이 크므로, 외부에서 1초 주기 tick 만 사용
    public onSecondTick(dt: number): void {
        this.elapsed += dt;

        for (const ev of this.schedule) {
            if (!ev.fired && this.elapsed >= ev.atSeconds) {
                this.fire(ev.type, ev.count);
                ev.fired = true;
            }
        }
    }

    private fire(type: DisasterType, count: number): void {
        for (let i = 0; i < count; i++) {
            // Pool 에서 재사용 — 직접 new 하지 않는다
            const d = DisasterPool.acquire(type);
            d.spawn(this.randomFieldPoint());
            this.active.push(d);
        }
    }
}
src/pool/ObjectPool.ts · conceptual TypeScript 
// 거인 / 이펙트 등 대량 인스턴스 — Pool 로 메모리 폭증 방지
export class ObjectPool<T extends IPoolable> {

    private available: T[] = [];
    private inUse: Set<T> = new Set();

    constructor(
        private factory: () => T,
        private initial: number = 16
    ) {
        for (let i = 0; i < initial; i++) {
            this.available.push(factory());
        }
    }

    public acquire(): T {
        const obj = this.available.pop() ?? this.factory();
        obj.onAcquire();          // 상태 초기화 — Awake 가 아니라 reuse
        this.inUse.add(obj);
        return obj;
    }

    public release(obj: T): void {
        if (!this.inUse.has(obj)) return;   // 중복 release 차단
        obj.onRelease();          // 비활성화 — Destroy 가 아니라 disable
        this.inUse.delete(obj);
        this.available.push(obj);
    }

    // 라운드 종료 시 일괄 회수 — 거인 / 이펙트 / 투사체 모두 한 번에
    public releaseAll(): void {
        this.inUse.forEach(obj => this.release(obj));
    }
}
src/multiplay/HostFailover.ts · conceptual TypeScript 
// Host 가 중간에 이탈하면 다른 client 가 권한을 인수해 게임을 이어간다
enum PeerRole { HOST = 0, CLIENT = 1 }

export class HostFailover {

    private role: PeerRole = PeerRole.CLIENT;
    private peers: Map<string, PeerInfo> = new Map();
    private lastHeartbeat: number = Date.now();

    public onHeartbeat(peerId: string, ts: number): void {
        const p = this.peers.get(peerId);
        if (p) p.lastSeen = ts;
        if (peerId === this.getHostId()) this.lastHeartbeat = ts;
    }

    // Host 가 N 초 동안 응답 없으면 이탈로 간주, 차순위 client 가 인수
    public tick(): void {
        const now = Date.now();
        if (now - this.lastHeartbeat > 5000) {
            this.electNewHost();
        }
    }

    private electNewHost(): void {
        // peer.id 사전순 최소 = 차순위 host — 모든 client 가 같은 결과 도출
        const sorted = [...this.peers.keys()].sort();
        const next = sorted[0];

        if (next === this.myId()) {
            this.role = PeerRole.HOST;
            GameState.resumeAsHost();   // 라운드 상태 복원 후 진행
        } else {
            RPC.listenTo(next);              // client 는 새 host 구독
        }
    }
}

Real problems, shipped solutions

프로덕션 환경에서 실제로 마주친 5가지 문제와 해결 과정. 출시 이후 라이브 이슈 대응을 포함합니다.

Host Failover
Host 가 중간에 나가면 전체 라운드가 중단되어 모든 client 가 튕김
solution
Host 이탈 감지 (heart-beat timeout) → 차순위 client 가 결정적 순서로 권한 인수. Client 측에서는 새 host 로의 RPC fallback 을 자동 처리하도록 설계.
Memory Spike
매 라운드 거인 / 이펙트 인스턴스를 새로 생성하니 메모리 폭증 + GC stall
solution
거인 · 이펙트 · 투사체에 Object Pool 도입. 인스턴스를 재사용하고 라운드 종료 시 일괄 회수. acquire / release 사이클로 메모리 안정화.
Exploit Reported
사용자가 앱을 백그라운드로 깔면 데미지 무효 — 끝까지 살아남는 버그 악용 신고
solution
무적존 패치 + Heart-beat 통신으로 백그라운드 상태 감지. 보안적 사고 — 문제 가능성을 미리 판단·대처할 수 있는 역량으로 정착.
Team Collaboration
다른 개발자가 코드를 읽기 어려워 협업 비용이 점점 커짐
