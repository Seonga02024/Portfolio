src/physics/ParabolaBall.ts TypeScript 
// 포물선 방정식 — 입력 강도 + 방향 → 시간에 따른 공 위치
//   x(t) = x₀ + vₓ · t
//   y(t) = y₀ + v_y · t − ½ · g · t²    ← 중력 적분
//   z(t) = z₀ + v_z · t

const GRAVITY = 9.81;  // m/s²

interface LaunchInput {
    origin: Vector3;          // 공 출발점
    target: Vector3;          // 사용자가 누른 코트 좌표
    power: number;            // 강도 게이지 [0..1]
}

function launchBall(input: LaunchInput): BallTrajectory {
    const dir = input.target.subtract(input.origin);
    const dist = dir.magnitude();

    // 강도에 따른 초기 속도 + 발사 각도 (45°가 최대 거리)
    const speed = lerp(8, 22, input.power);
    const angle = computeLaunchAngle(dist, speed);

    const v0 = {
        x: speed * Math.cos(angle) * dir.x / dist,
        y: speed * Math.sin(angle),
        z: speed * Math.cos(angle) * dir.z / dist,
    };

    // 포물선 위치 함수 — 매 프레임 t로 호출
    const positionAt = (t: number): Vector3 => ({
        x: input.origin.x + v0.x * t,
        y: input.origin.y + v0.y * t - 0.5 * GRAVITY * t * t,
        z: input.origin.z + v0.z * t,
    });

    // 착지 시각 — y(t) = floor 일 때의 t (이차방정식 해)
    const tLand = solveQuadratic(-0.5 * GRAVITY, v0.y, input.origin.y - COURT_Y);

    return { positionAt, tLand, v0 };
}
src/input/ScreenToCourt.ts TypeScript 
// 2D UI 입력 (화면 좌표) → 3D 월드 코트 좌표 변환
// 카메라가 비추는 코트 평면(y = 0)과 입력 광선의 교점을 구함

interface CourtPoint { x: number; z: number; }

function screenToCourt(
    screen: Vector2,        // 사용자가 누른 UI 좌표
    camera: Camera,           // 현재 카메라
): CourtPoint {
    // 1. 화면 좌표를 카메라 광선으로 변환 (Camera.ScreenToWorldPoint 패턴)
    const nearPt = camera.ScreenToWorldPoint({ x: screen.x, y: screen.y, z: camera.nearClip });
    const farPt  = camera.ScreenToWorldPoint({ x: screen.x, y: screen.y, z: camera.farClip });

    const ray = farPt.subtract(nearPt).normalized();

    // 2. 코트 평면 y = COURT_Y 와 광선의 교점
    //    t = (COURT_Y − origin.y) / ray.y
    const t = (COURT_Y - nearPt.y) / ray.y;

    if (t < 0) throw new Error("Ray does not hit court");

    // 3. 코트 평면 위 3D 좌표
    const world = {
        x: nearPt.x + ray.x * t,
        z: nearPt.z + ray.z * t,
    };

    // 4. 플레이어 진영 기준으로 코트 좌표 정규화 (반대편이면 좌우 반전)
    return normalizeToPlayerSide(world);
}

// 강도 게이지(2D) → 공 속도 스칼라(3D 발사 강도)
function gaugeToPower(gauge: number): number {
    return Math.min(Math.max(gauge, 0), 1);
}
src/rules/TennisScore.ts TypeScript 
// 테니스 점수 상태 기계 — 0/15/30/40, 듀스, 어드밴티지, 세트, 매치

type Point = "0" | "15" | "30" | "40";

enum GameState {
    NORMAL,        // 일반 진행
    DEUCE,         // 40-40
    ADVANTAGE_A,   // A가 어드밴티지
    ADVANTAGE_B,   // B가 어드밴티지
    GAME_OVER,
}

class TennisMatch {
    scoreA: Point = "0";
    scoreB: Point = "0";
    setsA: number = 0;
    setsB: number = 0;
    state: GameState = GameState.NORMAL;

    awardPoint(winner: "A" | "B"): void {
        switch (this.state) {
            case GameState.NORMAL:
                this.incrementNormal(winner);
                if (this.scoreA === "40" && this.scoreB === "40")
                    this.state = GameState.DEUCE;
                break;

            case GameState.DEUCE:
                this.state = winner === "A"
                    ? GameState.ADVANTAGE_A
                    : GameState.ADVANTAGE_B;
                break;

            case GameState.ADVANTAGE_A:
                if (winner === "A") this.winGame("A");
                else this.state = GameState.DEUCE;   // 듀스로 복귀
                break;

            case GameState.ADVANTAGE_B:
                if (winner === "B") this.winGame("B");
                else this.state = GameState.DEUCE;
                break;
        }
    }

    winGame(player: "A" | "B"): void {
        if (player === "A") this.setsA++; else this.setsB++;

        // 3판 중 2세트 승리 시 매치 종료
        if (this.setsA === 2 || this.setsB === 2) {
            this.state = GameState.GAME_OVER;
            return;
        }
        this.resetForNextSet();
    }
}
